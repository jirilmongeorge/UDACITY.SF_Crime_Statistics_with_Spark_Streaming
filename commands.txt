### Start Kafka / ZK
echo 'schema.registry.url=http://localhost:9092' >> /etc/kafka/connect-distributed.properties
nohup systemctl start confluent-zookeeper > startup.log &
nohup systemctl start confluent-kafka > startup.log &
sh start.sh

### Submitting Spark Kafka Ingest App
spark-submit --packages org.apache.spark:spark-sql-kafka-0-10_2.11:2.3.4 ingest_kafka_data.py


### Submitting Spark Variant with Triggers
spark-submit --packages org.apache.spark:spark-sql-kafka-0-10_2.11:2.3.4 trigger_variation.py

### Submitting Spark Progress Report
spark-submit --packages org.apache.spark:spark-sql-kafka-0-10_2.11:2.3.4 progress_reporter.py
