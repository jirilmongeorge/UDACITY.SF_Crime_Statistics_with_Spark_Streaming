### Start Kafka / ZK
echo 'schema.registry.url=http://localhost:9092' >> /etc/kafka/connect-distributed.properties

## Confluent Version
nohup systemctl start confluent-zookeeper > setup/startup_zK.log &
nohup systemctl start confluent-kafka > setup/startup_Kf.log &
sh start.sh

## apache Version

nohup /usr/bin/zookeeper-server-start config/zookeeper.properties > setup/startup_zK.log &
nohup /usr/bin/kafka-server-start config/server.properties > setup/startup_Kf.log &


### Submitting Spark Kafka Ingest App

### Submitting Spark Progress Report
spark-submit --packages org.apache.spark:spark-sql-kafka-0-10_2.11:2.3.4 progress_reporter.py
